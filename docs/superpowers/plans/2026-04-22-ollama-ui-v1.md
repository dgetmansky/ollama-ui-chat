# Ollama UI v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a one-page local Ollama diagnostic UI with a React frontend, an Express backend, JSON session storage, and one-command startup via `./run.sh`.

**Architecture:** The frontend is a Vite React SPA that talks only to a local Express backend. The backend owns session file persistence, Ollama proxying, request normalization, derived metrics, and abort handling. `./run.sh` starts both services in development mode so the app can be developed and run with one command.

**Tech Stack:** React, Vite, TypeScript, Express, Vitest, Testing Library, Supertest, Node.js

BACKEND_PID=$!

  sleep 1
done

until bash -lc '</dev/tcp/127.0.0.1/4173' >/dev/null 2>&1; do
  sleep 1
done

echo "Ollama UI GDP: http://127.0.0.1:4173"

wait

```ts
import type { DerivedMetrics } from "../types/session";

type OllamaStats = {
  total_duration?: number;
  load_duration?: number;
  prompt_eval_count?: number;
  prompt_eval_duration?: number;
  eval_count?: number;
  eval_duration?: number;
};

const nsToSec = (value?: number): number | null =>
  typeof value === "number" ? value / 1_000_000_000 : null;

const safeRate = (count?: number, duration?: number): number | null => {
  if (typeof count !== "number" || typeof duration !== "number" || duration <= 0) {
    return null;
  }

  return count / (duration / 1_000_000_000);
};

export const deriveMetrics = (stats: OllamaStats): DerivedMetrics => ({
  total_sec: nsToSec(stats.total_duration),
  load_sec: nsToSec(stats.load_duration),
  prompt_tokens_per_sec: safeRate(stats.prompt_eval_count, stats.prompt_eval_duration),
  eval_tokens_per_sec: safeRate(stats.eval_count, stats.eval_duration)
});
```

Create `backend/src/storage/sessionFiles.ts`:

```ts
import { mkdir, readdir, readFile, rm, writeFile } from "node:fs/promises";
import { join } from "node:path";

export const ensureSessionsDir = async (sessionsDir: string) => {
  await mkdir(sessionsDir, { recursive: true });
};

export const sessionPath = (sessionsDir: string, sessionId: string) =>
  join(sessionsDir, `${sessionId}.json`);

export const readSessionFile = async (sessionsDir: string, sessionId: string) =>
  readFile(sessionPath(sessionsDir, sessionId), "utf8");

export const writeSessionFile = async (sessionsDir: string, sessionId: string, text: string) => {
  await ensureSessionsDir(sessionsDir);
  await writeFile(sessionPath(sessionsDir, sessionId), text, "utf8");
};

export const deleteSessionFile = async (sessionsDir: string, sessionId: string) => {
  await rm(sessionPath(sessionsDir, sessionId), { force: true });
};

export const listSessionFiles = async (sessionsDir: string) => {
  await ensureSessionsDir(sessionsDir);
  return (await readdir(sessionsDir))
    .filter((name) => name.endsWith(".json"))
    .sort()
    .reverse();
};
```

Create `backend/src/storage/sessionStore.ts`:

```ts
import { randomUUID } from "node:crypto";
import { listSessionFiles, readSessionFile, writeSessionFile } from "./sessionFiles";
import type { DerivedMetrics, StoredSession } from "../types/session";

const emptyMetrics: DerivedMetrics = {
  total_sec: null,
  load_sec: null,
  prompt_tokens_per_sec: null,
  eval_tokens_per_sec: null
};

const createSessionId = () => {
  const stamp = new Date().toISOString().replace(/:/g, "-").replace(/\.\d{3}Z$/, "");
  return `${stamp}-${randomUUID()}`;
};

const createDefaultSession = (): StoredSession => {
  const now = new Date().toISOString();

  return {
    id: createSessionId(),
    created_at: now,
    updated_at: now,
    endpoint: "/api/chat",
    model: "",
    stream: true,
    request_options: {
      num_predict: 256,
      temperature: 0.7
    },
    messages: [],
    last_request: {},
    last_response: {},
    last_stats: {},
    derived_metrics: emptyMetrics,
    runtime: {
      last_request_id: null,
      last_status: "idle"
    }
  };
};

export const createSessionStore = ({ sessionsDir }: { sessionsDir: string }) => ({
  async create() {
    const session = createDefaultSession();
    await writeSessionFile(sessionsDir, session.id, JSON.stringify(session, null, 2));
    return session;
  },

  async list() {
    const files = await listSessionFiles(sessionsDir);

    return Promise.all(
      files.map(async (name) => JSON.parse(await readSessionFile(sessionsDir, name.replace(/\.json$/, ""))) as StoredSession)
    );
  }
});
```

  const [sessions, setSessions] = useState<SessionRecord[]>([]);
  const [activeSessionId, setActiveSessionId] = useState<string | null>(null);

  useEffect(() => {
    void Promise.all([api.listModels(), api.listSessions()]).then(([modelsResponse, sessionsResponse]) => {
      setModels(modelsResponse.models);
      setSessions(sessionsResponse.sessions);
      setActiveSessionId(sessionsResponse.sessions[0]?.id ?? null);
    });

  const refreshModels = async () => {
    const response = await api.listModels();
    setModels(response.models);
  };

    const response = await api.ping();
    setStatusText(response.status === "ok" ? "Backend reachable" : "Backend unavailable");

    setActiveSessionId(session.id);

    const session = await api.createSession();

    setActiveSessionId(session.id);

    setSessions((current) => current.filter((item) => item.id !== activeSessionId));

      return next?.id ?? null;

    last_response: {},
    last_stats: {},

export const createRunService = ({

          { role: "user", content: prompt },
          { role: "assistant", content: reply?.content ?? "" }

Update `backend/src/services/runService.ts`:

```ts
import { randomUUID } from "node:crypto";
import { buildChatPayload } from "../ollama/buildChatPayload";
import { buildGeneratePayload } from "../ollama/buildGeneratePayload";
import { deriveMetrics } from "../metrics/derivedMetrics";
import type { StoredSession } from "../types/session";

export const createRunService = ({
  runChat,
  runGenerate
}: {
  runChat: (payload: Record<string, unknown>) => Promise<Record<string, unknown>>;
  runGenerate: (payload: Record<string, unknown>) => Promise<Record<string, unknown>>;
}) => ({
  async run({
    session,
    prompt,
    endpoint,
    model,
    stream,
    request_options
  }: {
    session: StoredSession;
    prompt: string;
    endpoint: "/api/chat" | "/api/generate";
    model: string;
    stream: boolean;
    request_options: {
      num_predict: number;
      temperature: number;
    };
  }) {
    const preparedSession: StoredSession = {
      ...session,
      endpoint,
      model,
      stream,
      request_options,
      runtime: {
        last_request_id: randomUUID(),
        last_status: stream ? "running" : "completed"
      }
    };

    const payload =
      endpoint === "/api/generate"
        ? buildGeneratePayload({ session: preparedSession, prompt })
        : buildChatPayload({ session: preparedSession, prompt });

    const response =
      endpoint === "/api/generate"
        ? await runGenerate(payload)
        : await runChat(payload);

    const assistantText =
      endpoint === "/api/generate"
        ? (response.response as string | undefined) ?? ""
        : ((response.message as { content?: string } | undefined)?.content ?? "");

    const stats = {
      total_duration: response.total_duration,
      load_duration: response.load_duration,
      prompt_eval_count: response.prompt_eval_count,
      prompt_eval_duration: response.prompt_eval_duration,
      eval_count: response.eval_count,
      eval_duration: response.eval_duration
    };

    return {
      session: {
        ...preparedSession,
        updated_at: new Date().toISOString(),
        messages: [
          ...session.messages,
          { role: "user", content: prompt },
          { role: "assistant", content: assistantText }
        ],
        last_request: payload,
        last_response: response,
        last_stats: stats,
        derived_metrics: deriveMetrics(stats),
        runtime: {
          last_request_id: preparedSession.runtime.last_request_id,
          last_status: stream ? "completed" : "completed"
        }
      }
    };
  }
});
```

import { Router } from "express";

export const createRunRouter = ({
  getSession,
  saveSession,
  run
}: {
  getSession: (sessionId: string) => Promise<any>;
  saveSession: (session: any) => Promise<any>;
  run: (input: {
    session: any;
    prompt: string;
    endpoint: "/api/chat" | "/api/generate";
    model: string;
    stream: boolean;
    request_options: {
      num_predict: number;
      temperature: number;
    };
  }) => Promise<{ session: any }>;
}) => {
  const router = Router();

  router.post("/sessions/:id/run", async (request, response, next) => {
    try {
      const session = await getSession(request.params.id);
      const updated = await run({
        session,
        prompt: request.body.prompt,
        endpoint: request.body.endpoint,
        model: request.body.model,
        stream: request.body.stream,
        request_options: request.body.request_options
      });
      await saveSession(updated.session);
      response.json(updated);
    } catch (error) {
      next(error);
    }
  });

  router.post("/requests/:requestId/abort", (_request, response) => {
    response.status(202).json({ status: "accepted" });
  });

  return router;
};
